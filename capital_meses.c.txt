#include <stdio.h>

int main() {
    char modo = 'i';

    for (; modo != 'F'; ) {
        printf("\nDigite o modo (L - lavar, E - enxaguar, S - secar, F - fim): ");
        scanf(" %c", &modo); // espaço resolve o problema do ENTER

        switch (modo) {
            case 'L':
                printf(">> Máquina: LAVANDO...\n");
                break;
            case 'E':
                printf(">> Máquina: ENXAGUANDO...\n");
                break;
            case 'S':
                printf(">> Máquina: SECANDO...\n");
                break;
            case 'F':
                printf(">> Encerrando...\n");
                break;
            default:
                printf(">> Opção inválida!\n");
        }
    }

    printf("Finalizado\n");
    return 0;
}
