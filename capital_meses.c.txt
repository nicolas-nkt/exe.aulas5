#include <stdio.h>

int main() {
    int contador;
    float c; 
    float i; 
    int t;   

    printf("Digite seu capital: ");
    scanf("%f", &c);

    printf("Digite o juros (ex: 5 para 5%%): ");
    scanf("%f", &i);

    printf("Digite o tempo em meses: ");
    scanf("%d", &t);

    i = i / 100; // converte %

    printf("\nSeu rendimento é:\n");

    for(contador = 1; contador <= t; contador++) {
        c = c * (1 + i); // juros compostos

        printf("Mes %d: R$ %.2f\n", contador, c);
    }

    return 0;
}
